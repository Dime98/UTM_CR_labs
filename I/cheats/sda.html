 struct marfa *append(struct marfa *S, int *n)  (home)
 {
 	struct marfa *GS;
 	struct marfa rc;
	

 	addNew(&rc);


 	GS = (struct marfa*) realloc (S, (*n + 1) * sizeof(*GS));


 	if (!GS)
 		return 0;


 	GS[*n] = rc;
 	*n = *n + 1;
 	return GS;
 }





 struct marfa * preppend(struct marfa *S, int *n)  (home)
 {
 	int i;
 	struct marfa *FG;
 	struct marfa gxr;


 	addNew(&gxr);
	

 	FG = (struct marfa *) realloc (S, (*n + 1) * sizeof(*FG));


 	if (!FG)
 		return 0;


 	for (i = *n; i > 0; i--)
 		FG[i] = FG[i - 1];


 	FG[0] = gxr;
 	*n = *n + 1;
	

 	return FG;
 }

 struct marfa* insertAfter (struct marfa *S, int *n, int k)  (home)
 {
 	int i;
 	struct marfa *duke;
 	struct marfa cbr;
	

 	addNew(&cbr);


 	duke =( struct marfa*) realloc (S, (*n+1)*sizeof(*duke));
	

 	if (!duke)
 			return 0;
	
	

 	for (i=*n; i>k; i--)
 			duke[i] = duke[i - 1];
	

 	duke[k] = cbr;

 	return duke;
 }


 struct marfa* insertBefore (struct marfa *S, int *n, int k)  (home)

 	if (!duke)
 		return 0;
	

 	for (i = *n; i>= k; i--)
 		duke[i] = duke[i - 1];
