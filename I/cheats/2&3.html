 int create (int n)
 {
     marfa *c, *p;
     int i;
     for (i = 0; i < n; i++)
     {
         c = (marfa*) malloc (sizeof(*c));
         if(c == NULL) return 0;
         if(i == 0) 
             {
                 head = c;
                 p = c;
             }
         else 
             {
                 p  ->  next = c;
                 p = c;
             }
     }
     p  ->  next = head;
     return 1;
 }

 void swap(marfa *a,marfa *b)
 {
     marfa t,*na,*nb;
     na = a -> next;
     nb = b -> next;
     t = *a;
     *a = *b;
     *b = t;
     a -> next = na;
     b -> next = nb;
 }

 void deleteList (int id)


 {


     marfa *c = head,*p;


     int i = 0;



     while(c)


     {


         if(i == id - 1)


         {


             if(c == head)


             {


                 head = c -> next;


                 free(c);


                 break;


             } 


             else if(c -> next == NULL)


             {


                 p = head;


                 while(p)


                 {


                     if(p -> next == c)


                     {


                         free(c);


                         p -> next=NULL;


                         break;


                     }


                     p = p -> next;


                 }


             } 


             else {


                 p = head;


                 while(p)


                 {


                     if(p -> next == c)


                     {                   


                         p -> next = c -> next;


                         free(c);


                         break;


                     }


                     p = p -> next;


                 }


             }


         }


         i++;


         c = c -> next;


     }


 }
